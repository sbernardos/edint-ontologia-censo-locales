COM01: ¿Qué establecimientos comerciales existen en una división administrativa?
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT
       ?commercialSpace
       ?commercialSpaceLabel
       ?adminDivision
       ?adminDivisionLabel
WHERE {
  # Establecimientos comerciales como elementos de infraestructura
  ?commercialSpace a edintcom:CommercialSpace ;
                   a edintinf:InfrastructureElement ;
                   edintinf:belongsTo ?adminDivision .

  FILTER (?adminDivision IN (
    esadm:Municipio,
    esadm:Distrito,
	esadm:Barrio
  ))
  
  OPTIONAL {
    ?commercialSpace rdfs:label ?commercialSpaceLabel .
  }
  OPTIONAL {
    ?adminDivision rdfs:label ?adminDivisionLabel .
  }
}
ORDER BY ?commercialSpaceLabel

COM02: ¿Cuál es la distribución geográfica de los establecimientos por categoría de actividad CNAE o IAE?
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX skos:     <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>

SELECT
  ?adminDivision
  ?adminDivisionLabel
  ?activityCode
  ?activityLabel
  (COUNT(DISTINCT ?commercialSpace) AS ?numEstablishments)
WHERE {
  ?permit a edintcom:Permit ;
          edintcom:appliedTo ?commercialSpace ;
          ?activityProperty ?activityCode .

  FILTER (?activityProperty IN (
    edintcom:hasCNAECode,
    edintcom:hasIAECode
  ))

  ?commercialSpace a edintcom:CommercialSpace ;
                   a edintinf:InfrastructureElement ;
                   edintinf:belongsTo ?adminDivision .

  FILTER (?adminDivision IN (
    esadm:Municipio,
    esadm:Distrito,
	esadm:Barrio
  ))

  OPTIONAL { ?adminDivision rdfs:label ?adminDivisionLabel }
  OPTIONAL { ?activityCode skos:prefLabel ?activityLabel }
}
GROUP BY ?adminDivision ?adminDivisionLabel ?activityCode ?activityLabel
ORDER BY ?adminDivisionLabel ?activityLabel

COM03: ¿En qué estado están los comercios de una división administrativa concreta en un momento dado?
El estado del comercio en una fecha concreta se considera válido si existe una licencia aplicada al espacio comercial vigente en esa fecha.
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>
PREFIX schema:   <http://schema.org/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT DISTINCT
       ?commercialSpace
       ?commercialSpaceLabel
       ?spaceStatus
       ?spaceStatusLabel
WHERE {
  # Espacios comerciales de la división administrativa concreta
  ?commercialSpace a edintcom:CommercialSpace ;
                   a edintinf:InfrastructureElement ;
                   edintinf:belongsTo <URI-DIVISIÓN-ADM> ;
                   edintcom:hasCommercialSpaceStatus ?spaceStatus .

  # Licencia aplicada al espacio
  ?permit a edintcom:Permit ;
          edintcom:appliedTo ?commercialSpace ;
          schema:startDate ?startDate .

  OPTIONAL { ?permit schema:endDate ?endDate }

  # Momento temporal de interés
  FILTER (
    "FECHA-INICIO"^^xsd:dateTime >= ?startDate &&
    ( !BOUND(?endDate) ||
      "FECHA-FINAL"^^xsd:dateTime <= ?endDate )
  )

  OPTIONAL { ?commercialSpace rdfs:label ?commercialSpaceLabel }
  OPTIONAL { ?spaceStatus rdfs:label ?spaceStatusLabel }
}
ORDER BY ?spaceStatusLabel ?commercialSpaceLabel

COM05: ¿Cuántos comercios han abierto o cerrado en un periodo concreto en un distrito determinado?
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX schema:   <http://schema.org/>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT
  (SUM(?opened) AS ?numOpened)
  (SUM(?closed) AS ?numClosed)
WHERE {
  ?permit a edintcom:Permit ;
          edintcom:appliedTo ?commercialSpace ;
          schema:startDate ?startDate .

  OPTIONAL { ?permit schema:endDate ?endDate }

  ?commercialSpace a edintcom:CommercialSpace ;
                   a edintinf:InfrastructureElement ;
                   edintinf:belongsTo
                     <URI-DISTRITO> . #Valdría también cualquier URI de una división administrativa

  BIND(
    IF (
      ?startDate >= "FECHA-INICIO-PERIODO"^^xsd:dateTime &&
      ?startDate <= "FECHA-INICIO-PERIODO"^^xsd:dateTime,
      1, 0
    ) AS ?opened
  )

  BIND(
    IF (
      BOUND(?endDate) &&
      ?endDate >= "FECHA-FIN-PERIODO"^^xsd:dateTime &&
      ?endDate <= "FECHA-FIN-PERIODO"^^xsd:dateTime,
      1, 0
    ) AS ?closed
  )
}

COM06: ¿Qué zonas presentan mayor tasa de altas y bajas comerciales?
Considerando zona como distrito adeministrativo
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>
PREFIX esadm:    <http://vocab.linkeddata.es/datosabiertos/def/sector-publico/territorio#>
PREFIX schema:   <http://schema.org/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT
  ?zone
  ?zoneLabel
  (SUM(?opened) AS ?numOpened)
  (SUM(?closed) AS ?numClosed)
  (SUM(?opened + ?closed) AS ?turnover) # Tasa calculada como total de aperturas y cierres, por ejemplo
WHERE {
  ?permit a edintcom:Permit ;
          edintcom:appliedTo ?commercialSpace ;
          schema:startDate ?startDate .

  OPTIONAL { ?permit schema:endDate ?endDate }

  ?commercialSpace a edintcom:CommercialSpace ;
                   a edintinf:InfrastructureElement ;
                   edintinf:belongsTo ?zone .

  FILTER (?zone IN (
    esadm:Municipio,
    esadm:Distrito,
	esadm:Barrio
  ))
.

  OPTIONAL { ?zone rdfs:label ?zoneLabel }

  # Altas en el periodo
  BIND(
    IF (
      ?startDate >= "FECHA-INICIO-PERIODO"^^xsd:dateTime &&
      ?startDate <= "FECHA-INICIO-PERIODO"^^xsd:dateTime,
      1, 0
    ) AS ?opened
  )

  # Bajas en el periodo
  BIND(
    IF (
      BOUND(?endDate) &&
      ?endDate >= "FECHA-FIN-PERIODO"^^xsd:dateTime &&
      ?endDate <= "FECHA-FIN-PERIODO"^^xsd:dateTime,
      1, 0
    ) AS ?closed
  )
}
GROUP BY ?zone ?zoneLabel
ORDER BY DESC(?turnover)

COM07: ¿Qué licencias de actividad están asociadas a cada comercio?
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX schema:   <http://schema.org/>
PREFIX dcterms:  <http://purl.org/dc/terms/>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT DISTINCT
       ?commercialSpace
       ?commercialSpaceLabel
       ?permit
       ?fileCode
       ?permitType
       ?permitTypeLabel
       ?permitStatus
       ?permitStatusLabel
	   ?cnae
       ?cnaeLabel
       ?iae
       ?iaeLabel
       ?startDate
       ?endDate
WHERE {
  # Comercio
  ?commercialSpace a edintcom:CommercialSpace .

  # Licencia aplicada al comercio
  ?permit a edintcom:Permit ;
          edintcom:appliedTo ?commercialSpace .

  OPTIONAL { ?permit edintcom:fileCode ?fileCode }
  OPTIONAL { ?permit edintcom:hasCommercialPermitType ?permitType }
  OPTIONAL { ?permit edintcom:hasCommercialPermitStatus ?permitStatus }
  OPTIONAL { ?permit schema:startDate ?startDate }
  OPTIONAL { ?permit schema:endDate ?endDate }
  OPTIONAL { ?permit edintcom:hasCNAECode ?cnae )
  OPTIONAL { ?permit edintcom:hasIAECode ?iae )

  OPTIONAL { ?commercialSpace rdfs:label ?commercialSpaceLabel }
  OPTIONAL { ?permitType skos:prefLabel ?permitTypeLabel }
  OPTIONAL { ?permitStatus skos:prefLabel ?permitStatusLabel }
  OPTIONAL { ?cnae skos:prefLabel ?cnaeLabel }
  OPTIONAL { ?iae skos:prefLabel ?iaeLabel }

# Solo las licencias activas 
  FILTER (
    ?startDate <= NOW() &&
    ( !BOUND(?endDate) || ?endDate >= NOW() )
  )
}
ORDER BY ?commercialSpaceLabel ?startDate

PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>

SELECT
  ?permitType
  ?permitTypeLabel
  (COUNT(DISTINCT ?permit) AS ?numLicenciasVigentes)
WHERE {
  # Licencias con tipo
  ?permit a edintcom:Permit ;
          edintcom:hasCommercialPermitType ?permitType ;
          edintcom:appliedTo ?commercialSpace ;
          schema:startDate ?startDate .

  # Fecha de fin opcional
  OPTIONAL { ?permit schema:endDate ?endDate }

  # Comercios pertenecientes a la zona
  ?commercialSpace a edintcom:CommercialSpace ;
                   edintinf:belongsTo <URI-DIVISIÓN> . #Por ejemplo, <http://datos.ejemplo.es/recurso/distrito/Centro>

  # Filtro de licencias no caducadas
  FILTER (
    ?startDate <= NOW() &&
    ( !BOUND(?endDate) || ?endDate >= NOW() )
  )

  OPTIONAL {
    ?permitType rdfs:label ?permitTypeLabel .
  }
}
GROUP BY ?permitType ?permitTypeLabel

COM10.sparql
COM10: ¿Qué establecimientos comerciales disponen de autorización de ocupación de vía pública?
Hay que tener en cuenta se puede tratar de establecimientos que sean quioscos (y tengan licencia sin caducar) o de establecimientos que tengan terraza con licencia sin caducar
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX schema:   <http://schema.org/>
PREFIX skos:     <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT DISTINCT
       ?commercialSpace
       ?commercialSpaceLabel
	   ?infTypeLabel
       ?permit
       ?permitTypeLabel
       ?startDate
       ?endDate
WHERE {

  {
    ########################################################
    # CASO 1: QUIOSCOS con licencia vigente
    ########################################################
    ?permit a edintcom:Permit ;
            edintcom:appliedTo ?commercialSpace ;
            edintcom:hasCommercialPermitType ?permitType ;
            schema:startDate ?startDate .

    OPTIONAL { ?permit schema:endDate ?endDate }

    ?commercialSpace a edintcom:CommercialSpace ;
	                 edintinf:hasInfrastructureType ?inftype .

    ?infType skos:prefLabel ?infTypeLabel .

    FILTER (
      CONTAINS(LCASE(STR(?infTypeLabel)), "quiosco")
    )

    FILTER (
      ?startDate <= NOW() &&
      ( !BOUND(?endDate) || ?endDate >= NOW() )
    )
  }

  UNION

  {
    ########################################################
    # CASO 2: ESTABLECIMIENTOS CON TERRAZA y licencia vigente
    ########################################################
    ?commercialSpace a edintcom:CommercialSpace ;
                     edintcom:hasTerrace ?terrace .

    ?terrace a edintcom:Terrace .

    ?permit a edintcom:Permit ;
            edintcom:appliedTo ?terrace ;
            edintcom:hasCommercialPermitType ?permitType ;
            schema:startDate ?startDate .

    OPTIONAL { ?permit schema:endDate ?endDate }

    ?permitType skos:prefLabel ?permitTypeLabel .

    FILTER (
      ?startDate <= NOW() &&
      ( !BOUND(?endDate) || ?endDate >= NOW() )
    )
  }

  OPTIONAL {
    ?commercialSpace rdfs:label ?commercialSpaceLabel .
  }
}
ORDER BY ?commercialSpaceLabel

COM13.sparql
COM13: ¿Qué actividades comerciales existen en una zona determinada?
Sin tener en cuenta posibles correspondencias entre códigos CNAE e IAE
PREFIX edintcom: <http://vocab.linkeddata.es/datosabiertos/def/comercio/censo-locales#>
PREFIX edintinf: <http://vocab.linkeddata.es/datosabiertos/def/urbanismos-infraestructuras/infraestructura#>
PREFIX schema:   <http://schema.org/>
PREFIX skos:     <http://www.w3.org/2004/02/skos/core#>
PREFIX rdfs:     <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd:      <http://www.w3.org/2001/XMLSchema#>

SELECT
  ?activityType          # "CNAE" o "IAE"
  ?activityCode
  ?activityLabel
  (COUNT(DISTINCT ?permit) AS ?numLicenciasVigentes)
WHERE {
  # Licencias aplicadas a espacios comerciales de la zona
  ?permit a edintcom:Permit ;
          edintcom:appliedTo ?commercialSpace ;
          schema:startDate ?startDate .

  OPTIONAL { ?permit schema:endDate ?endDate }

  ?commercialSpace a edintcom:CommercialSpace ;
                   edintinf:belongsTo <URI-DIVISIÓN> . #Por ejemplo, <http://datos.ejemplo.es/recurso/distrito/Centro>

  # Licencias vigentes
  FILTER (
    ?startDate <= NOW() &&
    ( !BOUND(?endDate) || ?endDate >= NOW() )
  )

  # CNAE
  {
    ?permit edintcom:hasCNAECode ?activityCode .
    BIND("CNAE" AS ?activityType)
    OPTIONAL { ?activityCode skos:prefLabel ?activityLabel }
  }
  UNION
  # IAE
  {
    ?permit edintcom:hasIAECode ?activityCode .
    BIND("IAE" AS ?activityType)
    OPTIONAL { ?activityCode skos:prefLabel ?activityLabel }
  }
}
GROUP BY ?activityType ?activityCode ?activityLabel
ORDER BY ?activityType DESC(?numLicenciasVigentes) ?activityLabel
